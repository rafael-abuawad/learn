What you’re building

A stripped ERC20 with Yul for all hot paths: totalSupply, balanceOf, transfer, _mint, _burn. No permit, no Permit2, no metadata strings, no hooks — same patterns as Solady, fewer features.

Minimal contract

Copy this into a .sol file in your Foundry/Hardhat project. Annotated comments map each block to lessons 0001–0005.

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.4;

/// @notice Minimal ERC20 core — Yul patterns from Solady (lessons 0001–0006).
contract MinimalYulERC20 {
    // --- Namespaced storage (Lesson 0002) ---
    uint256 private constant _TOTAL_SUPPLY_SLOT = 0x05345cdf77eb68f44c;
    uint256 private constant _BALANCE_SLOT_SEED  = 0x87a211a2;

    // --- Precomputed constants (Lessons 0003–0004) ---
    uint256 private constant _TRANSFER_EVENT_SIGNATURE =
        0xddf252ad1be2c89b69c2b068fc378daa952ba7f163c4a11628f55a4df523b3ef;

    error TotalSupplyOverflow();
    error InsufficientBalance();

    event Transfer(address indexed from, address indexed to, uint256 amount);

    function totalSupply() public view returns (uint256 result) {
        assembly { result := sload(_TOTAL_SUPPLY_SLOT) }
    }

    function balanceOf(address owner) public view returns (uint256 result) {
        assembly {
            mstore(0x0c, _BALANCE_SLOT_SEED)
            mstore(0x00, owner)
            result := sload(keccak256(0x0c, 0x20))
        }
    }

    function transfer(address to, uint256 amount) public returns (bool) {
        assembly {
            // Phase 1–3: debit sender
            mstore(0x0c, _BALANCE_SLOT_SEED)
            mstore(0x00, caller())
            let fromSlot := keccak256(0x0c, 0x20)
            let fromBal := sload(fromSlot)
            if gt(amount, fromBal) {
                mstore(0x00, 0xf4d678b8) // InsufficientBalance()
                revert(0x1c, 0x04)
            }
            sstore(fromSlot, sub(fromBal, amount))

            // Phase 1–3: credit recipient
            mstore(0x00, to)
            let toSlot := keccak256(0x0c, 0x20)
            sstore(toSlot, add(sload(toSlot), amount))

            // Phase 4: emit
            mstore(0x20, amount)
            log3(0x20, 0x20, _TRANSFER_EVENT_SIGNATURE, caller(), shr(96, mload(0x0c)))
        }
        return true;
    }

    function _mint(address to, uint256 amount) internal {
        assembly {
            let before := sload(_TOTAL_SUPPLY_SLOT)
            let after := add(before, amount)
            if lt(after, before) {
                mstore(0x00, 0xe5cfe957) // TotalSupplyOverflow()
                revert(0x1c, 0x04)
            }
            sstore(_TOTAL_SUPPLY_SLOT, after)

            mstore(0x0c, _BALANCE_SLOT_SEED)
            mstore(0x00, to)
            let slot := keccak256(0x0c, 0x20)
            sstore(slot, add(sload(slot), amount))

            mstore(0x20, amount)
            log3(0x20, 0x20, _TRANSFER_EVENT_SIGNATURE, 0, shr(96, mload(0x0c)))
        }
    }

    function _burn(address from, uint256 amount) internal {
        assembly {
            mstore(0x0c, _BALANCE_SLOT_SEED)
            mstore(0x00, from)
            let slot := keccak256(0x0c, 0x20)
            let bal := sload(slot)
            if gt(amount, bal) {
                mstore(0x00, 0xf4d678b8)
                revert(0x1c, 0x04)
            }
            sstore(slot, sub(bal, amount))
            sstore(_TOTAL_SUPPLY_SLOT, sub(sload(_TOTAL_SUPPLY_SLOT), amount))

            mstore(0x00, amount)
            log3(0x00, 0x20, _TRANSFER_EVENT_SIGNATURE, shr(96, shl(96, from)), 0)
        }
    }
}

Exercises (in your own project)
