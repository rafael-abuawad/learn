Flow overview

  1. Revert if block.timestamp > deadline
  2. Clean owner and spender addresses
  3. Load nonce from storage (same slot pattern as nonces(owner))
  4. Build EIP-712 domain separator at free memory pointer m
  5. Build permit struct hash at m
  6. Hash 0x1901 || domainSeparator || structHash
  7. staticcall ecrecover precompile (address 1)
  8. Compare recovered signer to owner
  9. Increment nonce; store allowance; emit Approval

Domain separator (at free mem m)

mstore(m, _DOMAIN_TYPEHASH)
mstore(add(m, 0x20), nameHash)
mstore(add(m, 0x40), versionHash)
mstore(add(m, 0x60), chainid())
mstore(add(m, 0x80), address())
domainSeparator := keccak256(m, 0xa0)

ecrecover call

mstore(0x00, digest)      // keccak256("\x19\x01" || domain || structHash)
mstore(0x20, and(0xff, v))
mstore(0x40, r)
mstore(0x60, s)
let t := staticcall(gas(), 1, 0x00, 0x80, 0x20, 0x20)
if iszero(eq(mload(returndatasize()), owner)) { revert InvalidPermit }
sstore(nonceSlot, add(nonceValue, t))   // t=1 only if ecrecover succeeds

Nonce slot

mstore(0x0c, _NONCES_SLOT_SEED)
mstore(0x00, owner)
nonceSlot := keccak256(0x0c, 0x20)
